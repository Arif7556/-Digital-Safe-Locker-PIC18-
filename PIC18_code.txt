#include <p18f4520.h>
#include "LCD.h"


// Hardware Pins
#define LOCK_RELAY PORTDbits.RD2
#define BUZZER PORTDbits.RD3

// Function Prototypes
void system_init(void);
char keypad_getkey(void);
void eeprom_write(unsigned char address, unsigned char data);
unsigned char eeprom_read(unsigned char address);

// Global variables
unsigned char saved_pin[4];
unsigned char input_pin[4];
unsigned char wrong_attempts = 0;
unsigned int timeout_counter = 0;

void main(void)
{
    char key;
    int i, idx;
    
    system_init();
    lcd_init();
    
    // Read 4-digit PIN from EEPROM addresses 0 to 3
    for(i = 0; i < 4; i++)
    {
        saved_pin[i] = eeprom_read(i);
    }
    
    // If EEPROM is empty (0xFF), default the password to "1234" (ASCII values)
    if(saved_pin[0] == 0xFF)
    {
        eeprom_write(0, '1');
        eeprom_write(1, '2');
        eeprom_write(2, '3');
        eeprom_write(3, '4');
        saved_pin[0] = '1'; saved_pin[1] = '2'; saved_pin[2] = '3'; saved_pin[3] = '4';
    }

    lcd_cmd(0x80);
    lcd_string(" SAFE IS LOCKED ");

    while(1)
    {
        // --- STATE: LOCKED & AWAITING INPUT ---
        lcd_cmd(0x01); // Clear Screen
        lcd_cmd(0x80);
        lcd_string("Enter PIN:");
        
        idx = 0;
        while(idx < 4)
        {
            key = keypad_getkey();
            if(key != 0)
            {
                input_pin[idx] = key;
                lcd_cmd(0xC0 + idx);
                lcd_data('*'); // Mask display with asterisks
                
                // Quick beep for key press
                BUZZER = 1;
                for(i = 0; i < 500; i++); 
                BUZZER = 0;
                
                idx++;
            }
        }
        
        // Validate Entered PIN against Saved PIN
        if(input_pin[0] == saved_pin[0] && input_pin[1] == saved_pin[1] && 
           input_pin[2] == saved_pin[2] && input_pin[3] == saved_pin[3])
        {
            // --- STATE: ACCESS GRANTED ---
            wrong_attempts = 0;
            LOCK_RELAY = 1; // Unlock solenoid/relay
            
            lcd_cmd(0x01);
            lcd_string("ACCESS GRANTED");
            lcd_cmd(0xC0);
            lcd_string("1:Change  *:Lock");
            
            // Auto-lock Timeout counter loop (~10 Seconds)
            timeout_counter = 0;
            while(timeout_counter < 100)
            {
                key = keypad_getkey();
                
                if(key == '*') // Manual force lock
                {
                    break;
                }
                if(key == '1') // Secure Password Change Mode
                {
                    lcd_cmd(0x01);
                    lcd_string("New PIN:");
                    
                    idx = 0;
                    while(idx < 4)
                    {
                        key = keypad_getkey();
                        if(key != 0)
                        {
                            saved_pin[idx] = key;
                            lcd_cmd(0xC0 + idx);
                            lcd_data('*');
                            idx++;
                        }
                    }
                    
                    // Commit new password values to Internal EEPROM
                    for(i = 0; i < 4; i++)
                    {
                        eeprom_write(i, saved_pin[i]);
                    }
                    
                    lcd_cmd(0x01);
                    lcd_string("PIN CHANGED!");
                    for(i = 0; i < 20000; i++); // Display delay
                    break;
                }
                
                // Short structural delay loop
                for(i = 0; i < 1200; i++); 
                timeout_counter++;
            }
            
            // Auto-locking action triggered here
            LOCK_RELAY = 0; 
            lcd_cmd(0x01);
            lcd_string("LOCKING SAFE...");
            for(i = 0; i < 15000; i++);
        }
        else
        {
            // --- STATE: WRONG PASSWORD ---
            wrong_attempts++;
            lcd_cmd(0x01);
            lcd_string("WRONG PIN!");
            
            // Turn buzzer on for a warning duration
            BUZZER = 1;
            for(i = 0; i < 30000; i++);
            BUZZER = 0;
            
            // Security Lockout after 3 invalid attempts
            if(wrong_attempts >= 3)
            {
                lcd_cmd(0x01);
                lcd_string("SYSTEM LOCKED");
                lcd_cmd(0xC0);
                lcd_string("SECURITY ALARM");
                
                // Continuous penalty beep loop
                for(idx = 0; idx < 30; idx++)
                {
                    BUZZER = 1; for(i = 0; i < 5000; i++);
                    BUZZER = 0; for(i = 0; i < 15000; i++);
                }
                wrong_attempts = 0; // Reset counter post-penalty
            }
        }
    }
}

// System Hardware Initializations
void system_init(void)
{
        // Clean PORTD initial states
    ADCON1=0X0F;
    TRISB = 0xF0;         // Keypad Matrix Pins: RB0-RB3 Outputs(Rows), RB4-RB7 Inputs(Cols)
    INTCON2bits.RBPU = 0; // Enable PORTB Internal Pull-ups
}

// EEPROM Byte Write Logic
void eeprom_write(unsigned char address, unsigned char data)
{
    EEADR = address;
    EEDATA = data;
    EECON1bits.EEPGD = 0;
    EECON1bits.CFGS = 0;
    EECON1bits.WREN = 1;
    INTCONbits.GIE = 0;   // Turn off Interrupts
    EECON2 = 0x55;        // Obligatory Unlock Sequence
    EECON2 = 0xAA;
    EECON1bits.WR = 1;    // Trigger Write cycle
    INTCONbits.GIE = 1;   // Restore Interrupts
    while(EECON1bits.WR); // Wait until write operation finishes
    EECON1bits.WREN = 0;
}

// EEPROM Byte Read Logic
unsigned char eeprom_read(unsigned char address)
{
    EEADR = address;
    EECON1bits.EEPGD = 0;
    EECON1bits.CFGS = 0;
    EECON1bits.RD = 1;    // Trigger Read cycle
    return EEDATA;
}

// 4x4 Keypad Matrix Scanner Function
char keypad_getkey(void)
{
    int i;
    // Keypad layout definition array
    char keys[4][4] = {
        {'1','2','3','A'},
        {'4','5','6','B'},
        {'7','8','9','C'},
        {'*','0','#','D'}
    };
    
    // Row 0 Low Test
    PORTB = 0xFE; 
    for(i=0; i<10; i++);
    if(PORTBbits.RB4 == 0) { while(PORTBbits.RB4 == 0); return keys[0][0]; }
    if(PORTBbits.RB5 == 0) { while(PORTBbits.RB5 == 0); return keys[0][1]; }
    if(PORTBbits.RB6 == 0) { while(PORTBbits.RB6 == 0); return keys[0][2]; }
    if(PORTBbits.RB7 == 0) { while(PORTBbits.RB7 == 0); return keys[0][3]; }

    // Row 1 Low Test
    PORTB = 0xFD; 
    for(i=0; i<10; i++);
    if(PORTBbits.RB4 == 0) { while(PORTBbits.RB4 == 0); return keys[1][0]; }
    if(PORTBbits.RB5 == 0) { while(PORTBbits.RB5 == 0); return keys[1][1]; }
    if(PORTBbits.RB6 == 0) { while(PORTBbits.RB6 == 0); return keys[1][2]; }
    if(PORTBbits.RB7 == 0) { while(PORTBbits.RB7 == 0); return keys[1][3]; }

    // Row 2 Low Test
    PORTB = 0xFB; 
    for(i=0; i<10; i++);
    if(PORTBbits.RB4 == 0) { while(PORTBbits.RB4 == 0); return keys[2][0]; }
    if(PORTBbits.RB5 == 0) { while(PORTBbits.RB5 == 0); return keys[2][1]; }
    if(PORTBbits.RB6 == 0) { while(PORTBbits.RB6 == 0); return keys[2][2]; }
    if(PORTBbits.RB7 == 0) { while(PORTBbits.RB7 == 0); return keys[2][3]; }

    // Row 3 Low Test
    PORTB = 0xF7; 
    for(i=0; i<10; i++);
    if(PORTBbits.RB4 == 0) { while(PORTBbits.RB4 == 0); return keys[3][0]; }
    if(PORTBbits.RB5 == 0) { while(PORTBbits.RB5 == 0); return keys[3][1]; }
    if(PORTBbits.RB6 == 0) { while(PORTBbits.RB6 == 0); return keys[3][2]; }
    if(PORTBbits.RB7 == 0) { while(PORTBbits.RB7 == 0); return keys[3][3]; }

    return 0; // Return empty state if no button is pushed
}
